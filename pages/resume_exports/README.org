# File: tree-md
tree=$(tree -tf --noreport -I '*~' --charset ascii $1 |
       sed -e 's/| \+/*/g' -e 's/[|`]-\+/**/g' -e 's:\(* \)\(\(.*/\)\([^/]\+\)\):\1[[\2]]:g')
printf "* Project tree${tree}"
#+end_src
